create table user (
   id INT NOT NULL,
   username LONGTEXT NOT NULL,
   password LONGTEXT NOT NULL,
   citizen_id BIGINT NOT NULL,
   prefix VARCHAR(10) NOT NULL,
   firstname LONGTEXT NOT NULL,
   lastname LONGTEXT NOT NULL,
   firstname_en LONGTEXT NOT NULL,
   lastname_en LONGTEXT NOT NULL,
   grade INT NOT NULL,
   class INT NOT NULL,
   email LONGTEXT,
   isAdmin BOOLEAN DEFAULT false,
   isTeacher BOOLEAN DEFAULT false,
   isNewsReporter BOOLEAN DEFAULT false,
   isForumEditor BOOLEAN DEFAULT false,
   isRegistrator BOOLEAN DEFAULT false,
   isTimetableDesigner BOOLEAN DEFAULT false,
   isSubjectRegistrator BOOLEAN DEFAULT false,
   PRIMARY KEY ( id )
);

create table post (
   id INT NOT NULL AUTO_INCREMENT,
   title LONGTEXT NOT NULL,
   writer LONGTEXT NOT NULL,
   time DATETIME,
   isHidden BOOLEAN,
   article LONGTEXT NOT NULL,
   cover LONGTEXT,
   tags LONGTEXT,
   PRIMARY KEY ( id )
);

create table profile (
   id INT NOT NULL,
   tel VARCHAR(20),
   email MEDIUMTEXT,
   greetings LONGTEXT,
   profile LONGTEXT,
   background LONGTEXT,
   PRIMARY KEY ( id )
);

create table achievement (
   username LONGTEXT NOT NULL,
   id INT NOT NULL,
   page404 BOOLEAN DEFAULT false,
   betaTester BOOLEAN DEFAULT false,
   WebDevTycoon BOOLEAN DEFAULT false,
   the4thFloor BOOLEAN DEFAULT false,
   Xmas BOOLEAN DEFAULT false,
   PRIMARY KEY ( id )
);
